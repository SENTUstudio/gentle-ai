## When to Use

- When source files are hosted on SharePoint (common for business user uploads)
- When file delivery is via SharePoint folder structure
- When Lambda function approach is preferred over Glue (event-driven)
- When MSAL Graph API authentication is needed
- During `sdd-design` phase for SharePoint-based ETL approach

## Critical Patterns

### MSAL Authentication

Use `msal.ConfidentialClientApplication` for service-principal auth:

```python
import msal

TENANT_ID = os.getenv("TENANT_ID")
CLIENT_ID = os.getenv("CLIENT_ID")
CLIENT_SECRET = os.getenv("CLIENT_SECRET")
SCOPE = "https://graph.microsoft.com/.default"

def get_access_token() -> str:
    """Acquire token for service principal."""
    app = msal.ConfidentialClientApplication(
        CLIENT_ID,
        authority=f"https://login.microsoftonline.com/{TENANT_ID}",
        client_credential=CLIENT_SECRET,
    )
    result = app.acquire_token_for_client(scopes=[SCOPE])
    token = result.get("access_token")
    if not token:
        raise RuntimeError(f"MSAL auth failed: {result.get('error_description', result.get('error', 'unknown'))}")
    return token
```

### ENV Variables Required

These MUST be set in Lambda environment or local .env:

| Variable | Description |
|----------|-------------|
| `TENANT_ID` | Azure AD tenant ID |
| `CLIENT_ID` | Service principal app ID |
| `CLIENT_SECRET` | Service principal secret |
| `SITE_HOSTNAME` | SharePoint hostname (e.g. toyota.sharepoint.com) |
| `SITE_PATH` | SharePoint site path (e.g. /sites/toyota-chile) |
| `FOLDER_PATH` | Path to source folder in SharePoint |

### SharePoint File Download

Use Graph API to get download URL, then read directly:

```python
import urllib.parse

def get_drive_item_download_url(headers: dict, hostname: str, site_id: str, item_path: str) -> str:
    """Get pre-signed download URL for SharePoint item via driveItem metadata."""
    encoded_path = urllib.parse.quote(item_path, safe='/')
    url = f"https://graph.microsoft.com/v1.0/sites/{hostname},{site_id}/drive/root:/{encoded_path}"
    response = requests.get(url, headers=headers)
    response.raise_for_status()
    drive_item = response.json()
    return drive_item['@microsoft.graph.downloadUrl']
```

## Code Examples

### Lambda Handler Template

```python
# ============================================================================
# Lambda function: ETL_<table>_sharepoint
# source: SharePoint
# author: generated by gentle-ai
# ============================================================================

import os
import json
import requests
import pandas as pd
import awswrangler as wr
from datetime import datetime, date
import msal

# Configuration from ENV
TENANT_ID = os.getenv("TENANT_ID")
CLIENT_ID = os.getenv("CLIENT_ID")
CLIENT_SECRET = os.getenv("CLIENT_SECRET")
SITE_HOSTNAME = os.getenv("SITE_HOSTNAME", "toyota.sharepoint.com")
SITE_PATH = os.getenv("SITE_PATH", "/sites/toyota-chile")
FOLDER_PATH = os.getenv("FOLDER_PATH", "/Shared Documents/ETL/Input")
BUCKET = os.getenv("BUCKET", "toyota-chile-raw-data")

# File definitions
ARCHIVOS_A_PROCESAR = {
    "datos_ventas.csv": {
        "columns": ["id", "fecha", "producto", "cantidad", "monto"],
        "output_folder": "raw/ventas/",
        "read_options": {
            "sep": ";",
            "header": None,
            "encoding": "latin-1",
            "dtype": str,
        }
    },
    "datos_clientes.csv": {
        "columns": ["id", "nombre", "region", "segmento"],
        "output_folder": "raw/clientes/",
        "read_options": {
            "sep": ";",
            "header": None,
            "encoding": "latin-1",
            "dtype": str,
        }
    }
}

# ============================================================================
# AUTH
# ============================================================================
def get_access_token() -> str:
    """Acquire token for service principal."""
    app = msal.ConfidentialClientApplication(
        CLIENT_ID,
        authority=f"https://login.microsoftonline.com/{TENANT_ID}",
        client_credential=CLIENT_SECRET,
    )
    result = app.acquire_token_for_client(
        scopes=["https://graph.microsoft.com/.default"]
    )
    token = result.get("access_token")
    if not token:
        raise RuntimeError(f"MSAL auth failed: {result.get('error_description', result.get('error', 'unknown'))}")
    return token

# ============================================================================
# DOWNLOAD
# ============================================================================
import urllib.parse

SITE_HOSTNAME = os.getenv("SITE_HOSTNAME", "toyota.sharepoint.com")
SITE_PATH = os.getenv("SITE_PATH", "/sites/toyota-chile")

def resolve_site_id(headers: dict, hostname: str, site_path: str) -> str:
    """Resolve SharePoint site ID from hostname and path."""
    import urllib.parse
    encoded_path = urllib.parse.quote(site_path, safe='/:')
    url = f"https://graph.microsoft.com/v1.0/sites/{hostname}:{encoded_path}"
    response = requests.get(url, headers=headers)
    response.raise_for_status()
    return response.json()["id"]

def get_drive_items(headers: dict, site_id: str, folder_path: str) -> list:
    """List files in SharePoint folder."""
    encoded_path = urllib.parse.quote(folder_path, safe='/')
    url = f"https://graph.microsoft.com/v1.0/sites/{site_id}/drive/root:/{encoded_path}:/children"
    
    response = requests.get(url, headers=headers)
    items = response.json().get('value', [])
    return [item for item in items if item.get('name', '').endswith('.csv')]

def download_file(headers: dict, item_url: str) -> bytes:
    """Download file content from SharePoint."""
    response = requests.get(item_url, headers=headers)
    return response.content

# ============================================================================
# TRANSFORM
# ============================================================================
def transformar_dataframe(df: pd.DataFrame, nombre_archivo: str) -> pd.DataFrame:
    """Apply business transformations to DataFrame."""
    # Add metadata columns
    df["fecha_proceso"] = datetime.now().strftime("%Y-%m-%d")
    df["year"] = date.today().strftime("%Y")
    df["month"] = date.today().strftime("%m")
    df["day"] = date.today().strftime("%d")
    
    # Clean column names
    df.columns = [col.strip().lower().replace(" ", "_") for col in df.columns]
    
    return df

# ============================================================================
# HANDLER
# ============================================================================
def lambda_handler(event, context):
    """Main Lambda entry point."""
    # Auth
    access_token = get_access_token()
    headers = {"Authorization": f"Bearer {access_token}"}
    
    # Resolve site ID
    site_id = resolve_site_id(headers, SITE_HOSTNAME, SITE_PATH)
    
    # Get file list
    items = get_drive_items(headers, site_id, FOLDER_PATH)
    
    for item in items:
        nombre_archivo = item['name']
        if nombre_archivo not in ARCHIVOS_A_PROCESAR:
            continue
        
        info = ARCHIVOS_A_PROCESAR[nombre_archivo]
        
        # Download
        download_url = item['@microsoft.graph.downloadUrl']
        content = download_file(headers, download_url)
        
        # Read CSV
        from io import StringIO
        df = pd.read_csv(
            StringIO(content.decode('latin-1')),
            **info["read_options"]
        )
        df.columns = info["columns"]
        
        # Transform
        df_transformado = transformar_dataframe(df, nombre_archivo)
        
        # Write to S3
        s3_path = f"s3://{BUCKET}/{info['output_folder']}"
        wr.s3.to_parquet(
            df=df_transformado,
            path=s3_path,
            dataset=True,
            mode="overwrite",
            partition_cols=["year", "month", "day"]
        )
    
    return {"statusCode": 200, "body": json.dumps("Process complete")}
```
